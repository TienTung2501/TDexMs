# API Client

import { Callout } from 'nextra/components'

The frontend uses a centralized API client for all backend communication.

## Client Setup

```typescript
// lib/api.ts
const API_BASE = process.env.NEXT_PUBLIC_API_URL || 'http://localhost:3001';

async function apiFetch<T>(
  path: string,
  options?: RequestInit
): Promise<T> {
  const url = `${API_BASE}/v1${path}`;

  const res = await fetch(url, {
    ...options,
    headers: {
      'Content-Type': 'application/json',
      ...options?.headers,
    },
  });

  if (!res.ok) {
    const error = await res.json().catch(() => ({}));
    throw new ApiError(
      error.code || 'UNKNOWN',
      error.message || res.statusText,
      res.status
    );
  }

  return res.json();
}
```

### Intents

```typescript

const result = await apiFetch('/intents', {
  method: 'POST',
  body: JSON.stringify({
    creator: address,
    inputPolicyId: '',
    inputAssetName: '',
    inputAmount: '100000000',
    outputPolicyId: policyId,
    outputAssetName: assetName,
    minOutput: quote.minOutput,
    deadline: Math.floor(Date.now() / 1000) + 3600,
    partialFill: false,
    senderAddress: address,
    changeAddress: address,
  }),
});

// List user's intents
const intents = await apiFetch(`/intents?address=${address}&limit=20`);

// Cancel intent
const cancelResult = await apiFetch(`/intents/${intentId}`, {
  method: 'DELETE',
  body: JSON.stringify({ senderAddress: address }),
});
```

## WebSocket Client

```typescript
// lib/websocket.ts
const WS_URL = process.env.NEXT_PUBLIC_WS_URL || 'ws://localhost:3001';

export function createWSClient() {
  const ws = new WebSocket(`${WS_URL}/v1/ws`);

  return {
    subscribePrices: (pairs: string[]) => {
      ws.send(JSON.stringify({
        type: 'subscribe',
        channel: 'prices',
        params: { pairs },
      }));
    },

    subscribeIntent: (intentId: string) => {
      ws.send(JSON.stringify({
        type: 'subscribe',
        channel: 'intent',
        params: { intentId },
      }));
    },

    subscribePool: (poolId: string) => {
      ws.send(JSON.stringify({
        type: 'subscribe',
        channel: 'pool',
        params: { poolId },
      }));
    },

    onMessage: (handler: (msg: any) => void) => {
      ws.onmessage = (e) => handler(JSON.parse(e.data));
    },

    close: () => ws.close(),
  };
}
```

## Error Handling

```typescript
class ApiError extends Error {
  constructor(
    public code: string,
    message: string,
    public status: number,
  ) {
    super(message);
  }
}

// Usage in components
try {
  const result = await apiFetch('/intents', { method: 'POST', body });
  toast.success('Swap intent created!');
} catch (error) {
  if (error instanceof ApiError) {
    switch (error.code) {
      case 'INSUFFICIENT_FUNDS':
        toast.error('Insufficient balance for this swap');
        break;
      case 'RATE_LIMIT_EXCEEDED':
        toast.error('Too many requests. Please wait a moment.');
        break;
      default:
        toast.error(error.message);
    }
  }
}
```
