# Pipeline

import { Callout } from 'nextra/components'

The solver pipeline runs as a loop in `SolverEngine.runIteration()`.

## Iteration Flow

```typescript
async runIteration() {
  // Phase 1: Collect
  const rawIntents = await this.intentCollector.collect();

  // Phase 2: Filter
  const validIntents = this.filterAndValidate(rawIntents);
  if (validIntents.length === 0) return;

  // Phase 3: Route + Batch
  const routedIntents = await this.routeOptimizer.routeAll(validIntents);
  const batches = this.batchBuilder.build(routedIntents);

  // Phase 4: Netting + Settle
  for (const batch of batches) {
    const plan = NettingEngine.analyze(batch, poolReserves);
    const tx = await this.buildSettlementTx(batch, plan);
    await this.txSubmitter.submit(tx);
  }
}
```

## Stage 1: Intent Collection

**Module**: `IntentCollector` (171 lines)

The collector queries Blockfrost for all UTxOs at the escrow validator address:

```typescript
async collect(): Promise<RawIntent[]> {
  // 1. Get all UTxOs at escrow address
  const utxos = await this.blockfrost.getUtxos(this.escrowAddress);

  // 2. Parse each UTxO's inline datum (EscrowDatum — 12 fields)
  const intents = utxos
    .map(utxo => this.parseDatum(utxo))
    .filter(Boolean);

  // 3. Filter expired (past deadline)
  const now = Date.now();
  return intents.filter(i => i.deadline > now);
}
```

**Datum parsing** — The 12-field `Constr(0, [...])`:
```typescript
parseDatum(utxo) {
  const fields = utxo.datum.fields;
  return {
    creator:          fields[0],     // VerificationKeyHash
    inputPolicyId:    fields[1],
    inputAssetName:   fields[2],
    outputPolicyId:   fields[3],
    outputAssetName:  fields[4],
    minOutput:        BigInt(fields[5]),
    deadline:         BigInt(fields[6]),
    escrowToken:      fields[7],
    inputAmount:      BigInt(fields[8]),
    maxPartialFills:  Number(fields[9]),
    fillCount:        Number(fields[10]),
    remainingInput:   BigInt(fields[11]),
    // UTxO reference for spending
    utxoRef:          `${utxo.txHash}#${utxo.outputIndex}`,
  };
}
```

## Stage 2: Filter & Validate

**Module**: `SolverEngine.filterAndValidate()`

```typescript
filterAndValidate(intents: RawIntent[]): ValidIntent[] {
  return intents.filter(intent => {
    // Skip if already being processed this cycle
    if (this.processingSet.has(intent.utxoRef)) return false;

    // Skip if status is FILLING (mid-settlement)
    if (this.fillingSet.has(intent.escrowToken)) return false;

    // Skip stale intents from previous failed iterations
    if (intent.fillCount >= intent.maxPartialFills) return false;

    // Mark as processing to prevent double-processing
    this.processingSet.add(intent.utxoRef);
    return true;
  });
}
```

## Stage 3: Route Optimization

**Module**: `RouteOptimizer` (295 lines)

For each intent, find the best execution path:

```typescript
async findBestRoute(intent: ValidIntent): Promise<RoutedIntent> {
  const pool = await this.findPool(intent.inputAsset, intent.outputAsset);

  // Try full fill first
  const fullOutput = AmmMath.calculateSwapOutput(
    intent.remainingInput, pool.reserveIn, pool.reserveOut, pool.feeNumerator
  );

  if (fullOutput >= intent.proportionalMinOutput) {
    return { ...intent, route: 'DIRECT', fillAmount: intent.remainingInput };
  }

  // Fall back to partial fill (binary search)
  if (this.config.partialFillEnabled) {
    const maxFill = AmmMath.calculateMaxIntentFill(
      intent, pool, intent.proportionalMinOutput
    );

    if (maxFill >= intent.remainingInput / 10n) { // 10% min threshold
      return { ...intent, route: 'DIRECT_PARTIAL', fillAmount: maxFill };
    }
  }

  // Try multi-hop (via ADA)
  return this.tryMultiHop(intent);
}
```

## Stage 4: Batch Building

**Module**: `BatchBuilder` (118 lines)

```typescript
build(intents: RoutedIntent[]): Batch[] {
  // Group by pool
  const groups = this.groupByPool(intents);

  // Split groups exceeding constraints
  const batches: Batch[] = [];
  for (const group of groups) {
    if (group.length <= MAX_BATCH_SIZE) {
      batches.push(group);
    } else {
      // Split into sub-batches of MAX_BATCH_SIZE
      batches.push(...this.splitBatch(group));
    }
  }

  // Sort by urgency (earliest deadline first)
  return batches.sort((a, b) =>
    Math.min(...a.map(i => i.deadline)) - Math.min(...b.map(i => i.deadline))
  );
}
```

Constraints:

| Constraint | Value | Reason |
|-----------|-------|--------|
| `MAX_BATCH_SIZE` | 15 | Cardano TX size limit |
| `MAX_TX_CPU` | 14,000,000,000 | Plutus execution budget |
| `MAX_TX_MEM` | 10,000,000 | Plutus memory budget |

## Stage 5: Netting

**Module**: `NettingEngine` (273 lines)

For each batch, calculate optimal flow offsets:

```typescript
static analyze(batch: Batch, pool: PoolState): NettingPlan {
  // Separate by direction
  const aToB = batch.filter(i => i.direction === 'AToB');
  const bToA = batch.filter(i => i.direction === 'BToA');

  if (aToB.length === 0 || bToA.length === 0) {
    return this.singleDirectionPlan(batch, pool);
  }

  // Calculate gross flows
  const grossAtoB = sum(aToB.map(i => i.fillAmount));
  const grossBtoA = sum(bToA.map(i => i.fillAmount));

  // Convert to common denomination via spot price
  const grossBtoAInA = grossBtoA * pool.reserveA / pool.reserveB;

  // Net residual
  const netResidual = abs(grossAtoB - grossBtoAInA);
  const residualDirection = grossAtoB > grossBtoAInA ? 'AToB' : 'BToA';

  // Only residual hits AMM
  const ammOutput = AmmMath.calculateSwapOutput(
    netResidual, pool.reserveIn, pool.reserveOut, pool.feeNumerator
  );

  // Allocate outputs proportionally
  return {
    outputs: this.allocateOutputs(batch, ammOutput, grossAtoB, grossBtoA),
    nettingImprovement: 1 - netResidual / (grossAtoB + grossBtoAInA),
    residualDirection,
    residualAmount: netResidual,
  };
}
```

## Stage 6: Settlement

**Module**: `TxSubmitter` (170 lines) + `TxBuilder`

The settlement TX:

```typescript
async buildSettlementTx(batch: Batch, plan: NettingPlan) {
  const tx = lucid.newTx();

  // 1. Consume all escrow UTxOs (with Fill redeemer)
  for (const intent of batch) {
    tx.collectFrom([intent.utxo], fillRedeemer);
  }

  // 2. Consume pool UTxO (with Swap redeemer)
  tx.collectFrom([pool.utxo], swapRedeemer);

  // 3. Burn all intent tokens
  for (const intent of batch) {
    tx.mintAssets({ [intent.intentTokenUnit]: -1n }, burnRedeemer);
  }

  // 4. Send outputs to each user
  for (const { address, assets } of plan.outputs) {
    tx.payToAddress(address, assets);
  }

  // 5. Create new pool UTxO with updated reserves
  tx.payToContract(poolAddress, { inline: newPoolDatum }, newPoolValue);

  // 6. Set validity interval
  tx.validFrom(now).validTo(now + 300_000); // 5 min window

  // 7. Add collateral + sign
  tx.addSigner(solverPkh).complete();
}
```

The `TxSubmitter` uses a **singleton FIFO queue** shared by 3 bot types:

```
┌──────────────┐
│ Solver Bot   │──┐
├──────────────┤  │   ┌───────────────┐   ┌──────────────┐
│ ChainSync    │──┼──▶│ TX Queue      │──▶│ Blockfrost   │
│ Bot          │  │   │ (serial FIFO) │   │ submitTx()   │
├──────────────┤  │   └───────────────┘   └──────────────┘
│ Order        │──┘
│ Monitor Bot  │
└──────────────┘
```

This prevents UTxO contention — only one TX is in-flight at a time.

## Error Handling

```typescript
try {
  await this.txSubmitter.submit(tx);
} catch (error) {
  if (error.message.includes('UTxO already spent')) {
    // Normal during concurrency — retry next iteration
    this.processingSet.delete(intent.utxoRef);
  } else if (error.message.includes('SCRIPT_EXECUTION_FAILED')) {
    // Validator rejection — log and skip
    logger.error('Settlement TX rejected', { batch, error });
  } else {
    // Network/Blockfrost error — retry with backoff
    await this.retryWithBackoff(tx);
  }
}
```
