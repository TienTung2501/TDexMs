# WebSocket API

import { Callout } from 'nextra/components'

Real-time event streaming via WebSocket for prices, intent status changes, and pool updates.

## Connection

```
ws://localhost:3001/v1/ws
wss://your-api-domain.com/v1/ws
```

Upon successful connection, the server sends:

```json
{
  "type": "connected",
  "clientId": "ws_abc123",
  "timestamp": "2026-02-27T12:00:00Z"
}
```

<Callout type="info">
**Heartbeat:** The server pings every 30 seconds. Connections that fail to respond are automatically terminated.
</Callout>

---

## Channels

### Prices Channel

Subscribe to real-time price updates for token pairs.

**Subscribe:**
```json
{
  "type": "subscribe",
  "channel": "prices",
  "params": {
    "pairs": ["ADA/TOKEN", "ADA/USDC"]
  }
}
```

Use `"*"` for wildcard to receive all price updates:
```json
{
  "type": "subscribe",
  "channel": "prices",
  "params": { "pairs": ["*"] }
}
```

**Server Event:**
```json
{
  "type": "price",
  "data": {
    "pair": "ADA/TOKEN",
    "price": "0.5100",
    "change24h": "+2.5",
    "volume24h": "500000000",
    "timestamp": "2026-02-27T12:05:00Z"
  }
}
```

### Intent Channel

Subscribe to real-time updates for a specific intent.

**Subscribe:**
```json
{
  "type": "subscribe",
  "channel": "intent",
  "params": {
    "intentId": "intent_abc123"
  }
}
```

**Server Event:**
```json
{
  "type": "intentUpdate",
  "data": {
    "intentId": "intent_abc123",
    "status": "FILLED",
    "settlementTxHash": "def456...",
    "actualOutput": "49251234",
    "timestamp": "2026-02-27T12:05:00Z"
  }
}
```

## Unsubscribe

```json
{
  "type": "unsubscribe",
  "channel": "prices"
}
```

**Server Confirmation:**
```json
{
  "type": "unsubscribed",
  "channel": "prices"
}
```

## Client Example

```typescript
const ws = new WebSocket('ws://localhost:3001/v1/ws');

ws.onopen = () => {
  console.log('Connected');

  // Subscribe to price updates
  ws.send(JSON.stringify({
    type: 'subscribe',
    channel: 'prices',
    params: { pairs: ['*'] }
  }));

  // Subscribe to intent updates
  ws.send(JSON.stringify({
    type: 'subscribe',
    channel: 'intent',
    params: { intentId: 'intent_abc123' }
  }));
};

ws.onmessage = (event) => {
  const msg = JSON.parse(event.data);

  switch (msg.type) {
    case 'connected':
      console.log('Client ID:', msg.clientId);
      break;
    case 'price':
      console.log(`${msg.data.pair}: ${msg.data.price}`);
      break;
    case 'intentUpdate':
      console.log(`Intent ${msg.data.intentId}: ${msg.data.status}`);
      break;
    case 'poolUpdate':
      console.log(`Pool ${msg.data.poolId}: TVL ${msg.data.tvlAda}`);
      break;
    case 'error':
      console.error('WS Error:', msg.message);
      break;
  }
};

ws.onclose = () => console.log('Disconnected');
ws.onerror = (err) => console.error('WS Error:', err);
```

### React Hook Example

```typescript
import { useEffect, useRef, useCallback } from 'react';

export function useWebSocket(url: string) {
  const ws = useRef<WebSocket | null>(null);

  useEffect(() => {
    ws.current = new WebSocket(url);

    ws.current.onopen = () => {
      console.log('WebSocket connected');
    };

    return () => {
      ws.current?.close();
    };
  }, [url]);

  const subscribe = useCallback((channel: string, params: Record<string, any>) => {
    ws.current?.send(JSON.stringify({ type: 'subscribe', channel, params }));
  }, []);

  const unsubscribe = useCallback((channel: string) => {
    ws.current?.send(JSON.stringify({ type: 'unsubscribe', channel }));
  }, []);

  return { ws: ws.current, subscribe, unsubscribe };
}
```
